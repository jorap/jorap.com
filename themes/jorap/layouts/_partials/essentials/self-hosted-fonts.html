{{/* essentials/self-hosted-fonts — Build-time self-hosted Google Fonts. Replaces the legacy runtime injected to fonts.googleapis.com with a same-origin, fingerprinted, immutable CSS file plus pre-fetched woff2 binaries. Browsers stop hitting Google entirely (privacy + perf + no flash of unstyled font from a third-party connection). Algorithm: 1. Build the Google Fonts CSS URL from data/theme.json. 2. Fetch the CSS once at build with a desktop UA header — Google then returns modern woff2 sources segmented by unicode-range. 3. Pull every fonts.gstatic.com URL out of the response, download each binary as a Hugo resource, fingerprint+publish it under the site origin, and rewrite the CSS to reference the local path. 4. Emit the rewritten CSS as a fingerprinted resource and link it non-blocking (preload swap pattern). Error handling: resources.GetRemote returns nil on 404, otherwise fails the build on transport errors. Fonts are critical enough that a silent fallback would be worse than a loud build failure, so we don't use try. */}} {{- $pf := hugo.Data.theme.fonts.font_family.primary -}} {{- $sf := hugo.Data.theme.fonts.font_family.secondary -}} {{- $families := slice $pf -}} {{- if and $sf (ne $sf $pf) -}}{{- $families = $families | append $sf -}}{{- end -}} {{- $url := printf "https://fonts.googleapis.com/css2?family=%s&display=swap" (delimit $families "&family=") -}} {{- $opts := dict "headers" (dict "User-Agent" "Mozilla/5.0 (X11; Linux x86_64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/121.0.0.0 Safari/537.36" ) -}} {{- with resources.GetRemote $url $opts -}} {{- $css := .Content -}} {{- $remotes := findRE `https://fonts\.gstatic\.com/[^)\s]+` $css -}} {{- /* Deduplicate — Google often references the same woff2 across multiple @font-face blocks for different unicode-range slices. */ -}} {{- $seen := dict -}} {{- range $remotes -}} {{- $remote := . -}} {{- if not (index $seen $remote) -}} {{- $seen = merge $seen (dict $remote true) -}} {{- with resources.GetRemote $remote -}} {{/* resources.GetRemote publishes the binary without an extension, which would make Cloudflare/Netlify serve it as octet-stream and the browser would refuse to use it as a font. resources.Copy relocates it under /fonts/.woff2 so the static host infers font/woff2 from the suffix, and fingerprint pins the content-hash for the immutable cache header in static/_headers. */}} {{- $target := printf "fonts/%s" (path.Base $remote) -}} {{- $local := resources.Copy $target . | fingerprint "sha256" -}} {{- $css = replace $css $remote $local.RelPermalink -}} {{- end -}} {{- end -}} {{- end -}} {{- $bundle := resources.FromString "css/fonts.css" $css -}} {{- if hugo.IsProduction -}} {{- $bundle = $bundle | minify | fingerprint "sha256" -}} {{- end -}} {{/* Non-blocking stylesheet — paints with system fallback first, swaps in webfont when ready. font-display:swap inside the CSS guarantees the same behavior for clients that ignore the media="print" hack. */}} {{- else -}} {{ warnf "self-hosted-fonts: resources.GetRemote returned nil for %s (check font family names in data/theme.json)" $url }} {{- end -}}
