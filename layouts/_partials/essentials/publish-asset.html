{{/* Cache-bust a Hugo resource with content hashing (dev + production). Minify by default in production only. Params: - resource (required): Hugo resource object - minify (optional bool): minify before fingerprint (default: hugo.IsProduction) */}} {{- $asset := .resource -}} {{- $minify := .minify | default hugo.IsProduction -}} {{- if $minify -}} {{- $asset = $asset | minify -}} {{- end -}} {{- return ($asset | fingerprint) -}}
