{{/* Publish a Hugo asset with production fingerprinting for cache busting. Matches the main script/style bundles in themes/jorap/layouts/_partials/essentials/. Params: - src (required): asset path, e.g. "js/notes-graph.js" - as: "script" (default) or "style" - defer: include defer on scripts - async: include async on scripts */}} {{- $params := . -}} {{- $src := $params.src -}} {{- $as := $params.as | default "script" -}} {{- with resources.Get $src -}} {{- $asset := . -}} {{- if hugo.IsProduction -}} {{- $asset = $asset | minify | fingerprint -}} {{- end -}} {{- if eq $as "script" -}} {{- else if eq $as "style" -}} {{- end -}} {{- else -}} {{- warnf "fingerprint-asset: missing resource %q" $src -}} {{- end -}}
