{{- /* Shared TOC section links used by notes and blog TOCs. Params: Context, relatedPostsLimit, relatedNotesLimit (0 = skip), excludeSelfFromRandom */ -}} {{- $page := .Context -}} {{- $extras := slice -}} {{- $relatedLimit := .relatedPostsLimit | default 3 -}} {{- $relatedPosts := partial "blog/related-posts-lookup.html" (dict "Context" $page "Limit" $relatedLimit) | default slice -}} {{- if gt (len $relatedPosts) 0 -}} {{- $extras = $extras | append (dict "level" 2 "title" (T "related_posts") "id" "related-posts") -}} {{- end -}} {{- $relatedNotesLimit := .relatedNotesLimit | default 0 -}} {{- if gt $relatedNotesLimit 0 -}} {{- $relatedNotes := partial "notes/related-notes.html" (dict "Context" $page "Limit" $relatedNotesLimit) | default slice -}} {{- if gt (len $relatedNotes) 0 -}} {{- $extras = $extras | append (dict "level" 2 "title" (T "related_notes") "id" "related-notes") -}} {{- end -}} {{- end -}} {{- $randomPool := partial "notes/random-pool.html" $page -}} {{- if .excludeSelfFromRandom -}} {{- $randomPool = where $randomPool "url" "ne" $page.RelPermalink -}} {{- end -}} {{- if ge (len $randomPool) 1 -}} {{- $extras = $extras | append (dict "level" 2 "title" (T "random_blog_note") "id" "random-note") -}} {{- end -}} {{- return $extras -}}
