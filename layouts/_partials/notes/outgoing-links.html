{{- /* Wikilinks in the current note's raw body, resolved to pages. Returns slice of dict: Page, Label, Anchor. */ -}} {{- $page := . -}} {{- $raw := $page.RawContent -}} {{- $matches := findRE `\[\[[^\]|#]+(?:#[^\]|]+)?(?:\|[^\]]+)?\]\]` $raw -}} {{- $seen := dict -}} {{- $links := slice -}} {{- range $matches -}} {{- $inner := replaceRE `^\[\[|\]\]$` "" . -}} {{- $label := "" -}} {{- $target := $inner -}} {{- $anchor := "" -}} {{- if strings.Contains $inner "|" -}} {{- $parts := split $inner "|" -}} {{- $target = index $parts 0 | strings.TrimSpace -}} {{- $label = index $parts 1 | strings.TrimSpace -}} {{- end -}} {{- if strings.Contains $target "#" -}} {{- $hashParts := split $target "#" -}} {{- $target = index $hashParts 0 | strings.TrimSpace -}} {{- $anchor = index $hashParts 1 | strings.TrimSpace -}} {{- end -}} {{- $resolved := partial "notes/resolve-target.html" (dict "Target" $target) -}} {{- with $resolved.Page -}} {{- if ne .RelPermalink $page.RelPermalink -}} {{- if not (index $seen .RelPermalink) -}} {{- $seen = merge $seen (dict .RelPermalink true) -}} {{- if eq $label "" -}} {{- $label = .Title -}} {{- end -}} {{- $href := .RelPermalink -}} {{- if $anchor -}} {{- $href = printf "%s#%s" .RelPermalink ($anchor | anchorize) -}} {{- end -}} {{- $links = $links | append (dict "Page" . "Label" $label "Anchor" $anchor "Href" $href) -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- return $links -}}
