:= $label -}} {{- if not $text -}} {{- if $anchor -}} {{- $text = $anchor -}} {{- else -}} {{- $text = $target -}} {{- end -}} {{- end -}} {{- $link := printf `%s` $href $class .RelPermalink $text -}}
