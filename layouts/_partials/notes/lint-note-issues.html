{{- /* Build-time lint: warnf for broken wikilinks and unlinked mentions. Uses collect-note-issues.html; errorf when strict flags are set. */ -}} {{- $data := partialCached "notes/collect-note-issues.html" site "note-issues-v10" -}} {{- $strictLinks := site.Params.notes.strict_wikilinks | default false -}} {{- $strictUnlinked := site.Params.notes.strict_unlinked | default false -}} {{- range $data.notes -}} {{- $note := .Page -}} {{- range .broken -}} {{- $msg := printf "broken wikilink [[%s]] in %s" .target $note.File.Path -}} {{- if $strictLinks -}} {{- errorf $msg -}} {{- else -}} {{- warnf $msg -}} {{- end -}} {{- end -}} {{- range .pipe -}} {{- $msg := printf "pipe wikilink [[%s]] in %s — use [[%s]] only" .raw $note.File.Path .target -}} {{- if $strictLinks -}} {{- errorf $msg -}} {{- else -}} {{- warnf $msg -}} {{- end -}} {{- end -}} {{- range .unlinked -}} {{- $msg := printf "unlinked mention of %q (→ [[%s]]) in %s" .mention .suggestTitle $note.File.Path -}} {{- if $strictUnlinked -}} {{- errorf $msg -}} {{- else -}} {{- warnf $msg -}} {{- end -}} {{- end -}} {{- range .utility -}} {{- $msg := printf "utility wikilink [[%s]] in %s — use /notes/%s/" .target $note.File.Path .slug -}} {{- warnf $msg -}} {{- end -}} {{- range .relBroken -}} {{- $msg := printf "broken relationship (%s) [[%s]] in %s" .relType .target $note.File.Path -}} {{- warnf $msg -}} {{- end -}} {{- range .bodySection -}} {{- $msg := printf "body section ## %s still in markdown body: %s" .heading $note.File.Path -}} {{- warnf $msg -}} {{- end -}} {{- end -}}
