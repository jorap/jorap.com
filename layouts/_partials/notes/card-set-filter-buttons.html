{{- /* Filter chips for card sets with per-set counts. Params: cards (slice from collect-cards or cards-data), dataAttr ("data-notes-review-set" | "data-notes-cards-filter"). */ -}} {{- $cards := .cards -}} {{- $attr := .dataAttr | default "data-notes-review-set" -}} {{- $setNames := slice -}} {{- $setCounts := dict -}} {{- range $cards -}} {{- range .sets -}} {{- if not (in $setNames .) -}} {{- $setNames = $setNames | append . -}} {{- end -}} {{- $prev := index $setCounts . | default 0 -}} {{- $setCounts = merge $setCounts (dict . (add $prev 1)) -}} {{- end -}} {{- end -}} {{- $setNames = sort $setNames -}} {{- if eq $attr "data-notes-cards-filter" -}} {{- range $setNames -}} {{- $count := index $setCounts . | default 0 -}} {{- end -}} {{- else -}} {{- range $setNames -}} {{- $count := index $setCounts . | default 0 -}} {{- end -}} {{- end -}}
