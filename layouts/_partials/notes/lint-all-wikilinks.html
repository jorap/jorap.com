{{- /* Build-time wikilink lint. Called once from notes/list.html. warnf on broken links; errorf when notes.strict_wikilinks is true. */ -}} {{- $registry := partialCached "notes/link-registry.html" site "notes-registry-v1" -}} {{- $strict := site.Params.notes.strict_wikilinks | default false -}} {{- $broken := slice -}} {{- range $registry.notes -}} {{- $note := . -}} {{- $raw := $note.RawContent -}} {{- $fence := "```" -}} {{- $tick := "`" -}} {{- $raw = replaceRE (printf "(?s)%s[\\s\\S]*?%s" $fence $fence) "" $raw -}} {{- $raw = replaceRE (printf "%s[^%s\\n]+%s" $tick $tick $tick) "" $raw -}} {{- $matches := findRE `\[\[[^\]|#]+(?:#[^\]|]+)?(?:\|[^\]]+)?\]\]` $raw -}} {{- range $matches -}} {{- $inner := replaceRE `^\[\[|\]\]$` "" . -}} {{- $target := $inner -}} {{- if strings.Contains $inner "|" -}} {{- $target = index (split $inner "|") 0 | strings.TrimSpace -}} {{- end -}} {{- if strings.Contains $target "#" -}} {{- $target = index (split $target "#") 0 | strings.TrimSpace -}} {{- end -}} {{- if not (index $registry.byKey (lower $target)) -}} {{- $key := printf "%s|%s" $note.File.Path $target -}} {{- if not (in $broken $key) -}} {{- $broken = $broken | append $key -}} {{- $msg := printf "broken wikilink [[%s]] in %s" $target $note.File.Path -}} {{- if $strict -}} {{- errorf $msg -}} {{- else -}} {{- warnf $msg -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}}
