$start := 2 -}} {{- $end := 5 -}} {{- $ordered := true -}} {{- $listTag := cond $ordered "ol" "ul" -}} {{- $skipIds := slice "see-also" -}} {{- $html := "" -}} {{- $raw := partial "notes/assembled-content.html" $page -}} {{- $raw = replaceRE `(?s)\n## See also[\s\S]*$` "" $raw -}} {{- $headings := slice -}} {{- range $line := split $raw "\n" -}} {{- if findRE `^#{2,5}\s+` $line -}} {{- $level := len (index (findRE `^#{2,5}` $line) 0) -}} {{- if and (ge $level $start) (le $level $end) -}} {{- $title := replaceRE `^#{2,5}\s+` "" $line | strings.TrimSpace -}} {{- $title = replaceRE `\[\[([^\]|#]+)(?:#[^\]]+)?\]\]` "$1" $title -}} {{- $title = replaceRE `[*_]` "" $title -}} {{- $title = replaceRE `\{#[^}]+\}$` "" $title | strings.TrimSpace -}} {{- $title = $title | end -}} {{- end -}} {{- end -}} {{- end -}} {{- $extras := slice -}} {{- if gt (len (partial "notes/cards-for-page.html" $page)) 0 -}} {{- $extras = $extras | append (dict "level" 2 "title" "Flashcards" "id" "flashcards") -}} {{- end -}} {{- $outgoing := partial "notes/outgoing-links.html" $page | default slice -}} {{- if gt (len $outgoing) 0 -}} {{- $extras = $extras | append (dict   • %s
    • ` $items .level .id (.title | safeHTML) -}} {{- end -}} {{- $html = printf `
    <%s>%s
    ` $listTag $items $listTag -}} {{- end -}} {{- return (dict "html" $html "has" (ne $html "")) -}}
