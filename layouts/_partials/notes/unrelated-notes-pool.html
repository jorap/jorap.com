{{- /* All garden notes minus any that share a tag or category with Context. Returns a slice of dicts: url, title. */ -}} {{- $page := .Context -}} {{- $allNotes := partial "notes/garden-pages.html" $page -}} {{- $labels := $page.Params.tags | default slice -}} {{- $labels = $labels | append ($page.Params.categories | default slice) -}} {{- $pool := slice -}} {{- range $allNotes -}} {{- $include := true -}} {{- if gt (len $labels) 0 -}} {{- $noteLabels := .Params.tags | default slice -}} {{- $noteLabels = $noteLabels | append (.Params.categories | default slice) -}} {{- range $labels -}} {{- if in $noteLabels . -}} {{- $include = false -}} {{- end -}} {{- end -}} {{- end -}} {{- if $include -}} {{- $pool = $pool | append (dict "url" .RelPermalink "title" .Title) -}} {{- end -}} {{- end -}} {{- return $pool -}}
