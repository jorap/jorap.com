{{- /* Scan garden notes for broken wikilinks, pipe wikilinks, and unlinked title/alias mentions. Returns dict: byPath — map File.Path → { broken, pipe, unlinked, total } notes — slice of { Page, broken, pipe, unlinked, total } brokenCount, pipeCount, unlinkedCount, notesWithIssues */ -}} {{- $registry := partialCached "notes/link-registry.html" site "notes-registry-v2" -}} {{- $gardenNotes := partial "notes/garden-pages.html" (dict) -}} {{- $minLen := site.Params.notes.unlinked_min_length | default 6 | int -}} {{- $byPath := dict -}} {{- $allNotes := slice -}} {{- $brokenCount := 0 -}} {{- $pipeCount := 0 -}} {{- $unlinkedCount := 0 -}} {{- range $gardenNotes -}} {{- $note := . -}} {{- $path := $note.File.Path -}} {{- $raw := $note.RawContent -}} {{- $fence := "```" -}} {{- $tick := "`" -}} {{- $raw = replaceRE (printf "(?s)%s[\\s\\S]*?%s" $fence $fence) "" $raw -}} {{- $raw = replaceRE (printf "%s[^%s\\n]+%s" $tick $tick $tick) "" $raw -}} {{- $raw = replaceRE `(?s)]*id="see-also"[^>]*>See also.*$` "" $raw -}} {{- $raw = replaceRE `(?s)## See also.*$` "" $raw -}} {{- $broken := slice -}} {{- $brokenKeys := dict -}} {{- $matches := findRE `\[\[[^\]|#]+(?:#[^\]|]+)?\]\]` $raw -}} {{- range $matches -}} {{- $inner := replaceRE `^\[\[|\]\]$` "" . -}} {{- $target := $inner -}} {{- if strings.Contains $target "#" -}} {{- $target = index (split $target "#") 0 | strings.TrimSpace -}} {{- end -}} {{- $target = replace $target "\u2019" "'" -}} {{- $target = replace $target "\u2018" "'" -}} {{- if not (index $registry.byKey (lower $target)) -}} {{- $key := lower $target -}} {{- if not (index $brokenKeys $key) -}} {{- $brokenKeys = merge $brokenKeys (dict $key true) -}} {{- $broken = $broken | append (dict "target" $target "raw" $inner) -}} {{- end -}} {{- end -}} {{- end -}} {{- $pipe := slice -}} {{- $pipeKeys := dict -}} {{- $pipeMatches := findRE `\[\[[^\]]+\|[^\]]+\]\]` $raw -}} {{- range $pipeMatches -}} {{- $inner := replaceRE `^\[\[|\]\]$` "" . -}} {{- $key := lower $inner -}} {{- if not (index $pipeKeys $key) -}} {{- $pipeKeys = merge $pipeKeys (dict $key true) -}} {{- $target := $inner -}} {{- if strings.Contains $inner "|" -}} {{- $target = index (split $inner "|") 0 | strings.TrimSpace -}} {{- end -}} {{- $pipe = $pipe | append (dict "raw" $inner "target" $target) -}} {{- end -}} {{- end -}} {{- $unlinked := slice -}} {{- $unlinkedKeys := dict -}} {{- range $other := $gardenNotes -}} {{- if eq $other.RelPermalink $note.RelPermalink -}} {{- continue -}} {{- end -}} {{- $needles := slice $other.Title -}} {{- with $other.Params.aliases -}} {{- $needles = union $needles . -}} {{- end -}} {{- range $needles -}} {{- if lt (len .) $minLen -}} {{- continue -}} {{- end -}} {{- $words := split . " " -}} {{- $multiWord := gt (len $words) 1 -}} {{- $escaped := replaceRE `([.*+?^${}()|[\]\\])` `\\$1` . -}} {{- $wikilinkPattern := printf `(?i)\[\[%s(?:#[^\]|]+)?\]\]` $escaped -}} {{- if findRE $wikilinkPattern $raw -}} {{- continue -}} {{- end -}} {{- if $multiWord -}} {{- if not (findRE (printf `(?i)%s` $escaped) $raw) -}} {{- continue -}} {{- end -}} {{- $wordPattern := printf `(?i)(?:^|[^[\w])%s(?:[^]\w]|$)` $escaped -}} {{- if not (findRE $wordPattern $raw) -}} {{- continue -}} {{- end -}} {{- else -}} {{- if not (findRE (printf `%s` $escaped) $raw) -}} {{- continue -}} {{- end -}} {{- $wordPattern := printf `(?:^|[^[\wA-Z])%s(?:[^]\w]|$)` $escaped -}} {{- if not (findRE $wordPattern $raw) -}} {{- continue -}} {{- end -}} {{- end -}} {{- $ukey := printf "%s|%s" $path (lower .) -}} {{- if not (index $unlinkedKeys $ukey) -}} {{- $unlinkedKeys = merge $unlinkedKeys (dict $ukey true) -}} {{- $unlinked = $unlinked | append (dict "mention" . "suggestTitle" $other.Title "suggestPage" $other) -}} {{- end -}} {{- end -}} {{- end -}} {{- if or (gt (len $broken) 0) (gt (len $pipe) 0) (gt (len $unlinked) 0) -}} {{- $total := add (add (len $broken) (len $pipe)) (len $unlinked) -}} {{- $byPath = merge $byPath (dict $path (dict "broken" $broken "pipe" $pipe "unlinked" $unlinked "total" $total)) -}} {{- $allNotes = $allNotes | append (dict "Page" $note "broken" $broken "pipe" $pipe "unlinked" $unlinked "total" $total) -}} {{- $brokenCount = add $brokenCount (len $broken) -}} {{- $pipeCount = add $pipeCount (len $pipe) -}} {{- $unlinkedCount = add $unlinkedCount (len $unlinked) -}} {{- end -}} {{- end -}} {{- return (dict "byPath" $byPath "notes" $allNotes "brokenCount" $brokenCount "pipeCount" $pipeCount "unlinkedCount" $unlinkedCount "notesWithIssues" (len $allNotes) ) -}}
