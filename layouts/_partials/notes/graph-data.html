(findRE (printf `\[\[%s(?:#[^\]|]+)?(?:\|[^\]]+)?\]\]` .) $raw) -}} {{- $edges = $edges | append (dict "source" $fromId "target" $to.RelPermalink) -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- $inDeg := dict -}} {{- $outDeg := dict -}} {{- $totalDeg := dict -}} {{- range $nodes -}} {{- $inDeg = merge $inDeg (dict .id 0) -}} {{- $outDeg = merge $outDeg (dict .id 0) -}} {{- $totalDeg = merge $totalDeg (dict .id 0) -}} {{- end -}} {{- range $edges -}} {{- $ov := index $outDeg .source | default 0 -}} {{- $iv := index $inDeg .target | default 0 -}} {{- $outDeg = merge $outDeg (dict .source (add $ov 1)) -}} {{- $inDeg = merge $inDeg (dict .target (add $iv 1)) -}} {{- $ts := index $totalDeg .source | default 0 -}} {{- $tt := index $totalDeg .target | default 0 -}} {{- $totalDeg = merge $totalDeg (dict .source (add $ts 1)) -}} {{- $totalDeg = merge $totalDeg (dict .target (add $tt 1)) -}} {{- end -}} {{- $enriched := slice -}} {{- range $nodes -}} {{- $in := index $inDeg .id | default 0 -}} {{- $out := index $outDeg .id | default 0 -}} {{- $total := index $totalDeg .id | default 0 -}} {{- $enriched = $enriched | append (merge . (dict "inDegree" $in "outDegree" $out "totalDegree" $total "orphan" (eq $in 0) "hub" (or (ge $out $hubOutThreshold) (ge $total $hubTotalThreshold)) "deadEnd" (eq $out 0) )) -}} {{- end -}} {{- $maxDeg := -1 -}} {{- range $totalDeg -}} {{- if gt . $maxDeg -}} {{- $maxDeg = . -}} {{- end -}} {{- end -}} {{- $nearMax := slice -}} {{- range $id, $count := $totalDeg -}} {{- if ge $count (sub $maxDeg 1) -}} {{- $nearMax = $nearMax | append $id -}} {{- end -}} {{- end -}} {{- $focusId := "" -}} {{- $bestOut := -1 -}} {{- range $nearMax -}} {{- $oid := . -}} {{- $out := index $outDeg $oid | default 0 -}} {{- if gt $out $bestOut -}} {{- $bestOut = $out -}} {{- $focusId = $oid -}} {{- end -}} {{- end -}} {{- $hubTitles := slice "Maps of Content" "PKM" "Getting Started" -}} {{- range $hubTitles -}} {{- $hubTitle := . -}} {{- range $enriched -}} {{- if and (in $nearMax .id) (eq .title $hubTitle) -}} {{- if eq (index $outDeg .id | default 0) $bestOut -}} {{- $focusId = .id -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- if eq $focusId "" -}} {{- $focusId = index $nearMax 0 -}} {{- end -}} {{- return (dict "nodes" $enriched "edges" $edges "focus" $focusId "mode" "global") -}}
