{{- /* Table of contents for blog posts — body headings from Hugo's TOC, plus page sections (related posts, related notes, random note) when present. */ -}} {{- $page := . -}} {{- $bodyToc := $page.TableOfContents -}} {{- $emptyBody := not (findRE `<(ol|ul)` $bodyToc) -}} {{- $html := "" -}} {{- $relatedLimit := site.Params.blog.related_posts_limit | default 3 -}} {{- $relatedNotesLimit := site.Params.blog.related_notes_limit | default 5 -}} {{- $extras := partial "toc-related-extras.html" (dict "Context" $page "relatedPostsLimit" $relatedLimit "relatedNotesLimit" $relatedNotesLimit "excludeSelfFromRandom" false ) -}} {{- $extraItems := "" -}} {{- range $extras -}} {{- $extraItems = printf `%s
  • %s
    • ` $extraItems .id (.title | safeHTML) -}} {{- end -}} {{- if not $emptyBody -}} {{- if $extraItems -}} {{- $html = replaceRE `(\s*\s*)$` (printf "%s$1" $extraItems) $bodyToc 1 -}} {{- else -}} {{- $html = $bodyToc -}} {{- end -}} {{- else if $extraItems -}} {{- $html = printf `
      %s
    ` $extraItems -}} {{- end -}} {{- return (dict "html" $html "has" (ne $html "")) -}}
